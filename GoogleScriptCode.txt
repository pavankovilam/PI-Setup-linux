function doPost(e) {
  try {
    // e is the data recived from the Script API
    var ss = SpreadsheetApp.openById("1KUtLAX_H2KigvJ-obN14yccWkKGgvBWp1GyMT-CEdSA"); //Google Sheet ID
    var data = JSON.parse(e.postData.contents); //Converting JSON into JavaScript Object
    var sheet; //Pre Declaration of Variable as we don't know the Destination Sub-Sheet 
    // Selecting Sub-Sheet based on data recived from API
    if (data.deviceType == "EDC") {
      sheet = ss.getSheetByName("EDC8100"); //for EDC
    } else if (data.deviceType == "Envion5") {
      sheet = ss.getSheetByName("Envion5"); //For Envion5
    }
    else if (data.deviceType == "MM") {
      sheet= ss.getSheetId(1630731435); //For Only Memory Cards
    }
    else {
      return ContentService.createTextOutput("Error: Invalid device type or sheet not found")
                           .setMimeType(ContentService.MimeType.TEXT); // if some junk data comes from API Terminate here to reduce load on server
    }

    var cleanMac = data.mac.toUpperCase(); // Converting MAC to upper case for validation
    var lastRow = sheet.getLastRow(); // loading last row for adding 
    
    // Searching MAC
    var existingRowIndex = -1; //To store row inndex
    var assignedSerial = null; // intitially we did not assign a serial number so we will assumme it as null

    if (lastRow > 0) { 
      var macColumnValues = sheet.getRange(1, 4, lastRow, 1).getValues();  //getRange syntax(StartingRow, ColumnNumber(d=4), LastRow, NummberOfColumns) To store all MAC ids in a list serially
      for (var i = 0; i < macColumnValues.length; i++) { //looping through all MACs
        if (macColumnValues[i][0] === cleanMac) { // If MAC matches it enters, uodates existingRowIndex and break loop
          existingRowIndex = i + 1; // balancing Array index and row index as rows start with 1 and arrays start with 0
          break;
        }
      }
    }

    if (existingRowIndex !== -1) {
      //Update existing row
      assignedSerial = sheet.getRange(existingRowIndex, 2).getValue(); // Get existing Serial from Col 2
      
      // Update AnyDesk (Col 5) and TeamViewer (Col 6)
      sheet.getRange(existingRowIndex, 5).setValue(data.anydesk || "N/A"); //update anydesk and teamveiwer IDs
      sheet.getRange(existingRowIndex, 6).setValue(data.teamviewer || "N/A");
      //For Referencce Update the timestamp also
      sheet.getRange(existingRowIndex, 1).setValue(new Date()); 
      
      return ContentService.createTextOutput("Success - Updated Existing Serial: " + assignedSerial)
                           .setMimeType(ContentService.MimeType.TEXT);

    } else {
      // MAC NOT found: Create new entry
      var newSerial = 1;
      if (lastRow > 0) {
        var lastSerial = sheet.getRange(lastRow, 2).getValue();
        newSerial = (typeof lastSerial === 'number') ? lastSerial + 1 : 1;
      }

      sheet.appendRow([
        new Date(), 
        newSerial,
        data.deviceType, 
        cleanMac, 
        data.anydesk || "N/A", 
        data.teamviewer || "N/A"
      ]);
      
      return ContentService.createTextOutput("Success - Assigned New Serial: " + newSerial)
                           .setMimeType(ContentService.MimeType.TEXT);
    }
                           
  } catch (f) {
    return ContentService.createTextOutput("Error: " + f.toString())
                         .setMimeType(ContentService.MimeType.TEXT);
  }
}

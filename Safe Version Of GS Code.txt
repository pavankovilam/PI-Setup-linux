function doPost(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var data = JSON.parse(e.postData.contents);
    
    var sheet;
    // Determine which sheet to use based on deviceType
    if (data.deviceType == "EDC") {
      sheet = ss.getSheetByName("EDC8100");
    } else if (data.deviceType == "Envion5") {
      sheet = ss.getSheetByName("Envion5");
    }

    if (!sheet) {
      return ContentService.createTextOutput("Error: Invalid device type or sheet not found")
                           .setMimeType(ContentService.MimeType.TEXT);
    }

    var cleanMac = data.mac ? data.mac.toUpperCase() : "N/A";
    var lastRow = sheet.getLastRow();
    
    // --- NEW LOGIC: Search for existing MAC ---
    var existingRowIndex = -1;
    var assignedSerial = null;

    if (lastRow > 0) {
      // Assuming MAC is in Column 4 (D) based on your appendRow order
      var macColumnValues = sheet.getRange(1, 4, lastRow, 1).getValues(); 
      for (var i = 0; i < macColumnValues.length; i++) {
        if (macColumnValues[i][0] === cleanMac) {
          existingRowIndex = i + 1; // Rows are 1-indexed
          break;
        }
      }
    }

    if (existingRowIndex !== -1) {
      // MAC found: Update existing row
      assignedSerial = sheet.getRange(existingRowIndex, 2).getValue(); // Get existing Serial from Col 2
      
      // Update AnyDesk (Col 5) and TeamViewer (Col 6)
      sheet.getRange(existingRowIndex, 5).setValue(data.anydesk || "N/A");
      sheet.getRange(existingRowIndex, 6).setValue(data.teamviewer || "N/A");
      // Optional: Update the timestamp as well
      sheet.getRange(existingRowIndex, 1).setValue(new Date()); 
      
      return ContentService.createTextOutput("Success - Updated Existing Serial: " + assignedSerial)
                           .setMimeType(ContentService.MimeType.TEXT);

    } else {
      // MAC NOT found: Create new entry
      var newSerial = 1;
      if (lastRow > 0) {
        var lastSerial = sheet.getRange(lastRow, 2).getValue();
        newSerial = (typeof lastSerial === 'number') ? lastSerial + 1 : 1;
      }

      sheet.appendRow([
        new Date(), 
        newSerial,
        data.deviceType, 
        cleanMac, 
        data.anydesk || "N/A", 
        data.teamviewer || "N/A"
      ]);
      
      return ContentService.createTextOutput("Success - Assigned New Serial: " + newSerial)
                           .setMimeType(ContentService.MimeType.TEXT);
    }
                           
  } catch (f) {
    return ContentService.createTextOutput("Error: " + f.toString())
                         .setMimeType(ContentService.MimeType.TEXT);
  }
}
